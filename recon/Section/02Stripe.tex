\begin{frame}\frametitle{相对反射率校正}
计算步骤：
\begin{enumerate}
  \item<1-> 保持~$R$~通道不变，计算~$G$~通道和~$B$~通道对~$R$~通道的反射率
  \item<2-> 利用 Earther Mover's Distance，计算~$G$~通道灰度级~$i$~到~$R$~通道灰度~$j$~的流~$f_{ij}$~
  \item<3-> $G$~通道灰度级~$i$~的新灰度为：
  \begin{equation} \label{eq:ratio}
     new(i)=\frac{\sum_{j=1}^{255}{(f_{ij}j)}}{\sum_{j=1}^{255}{f_{ij}}}
     \nonumber
  \end{equation}
\end{enumerate}
\end{frame}

\subsection{峰值定位}
\begin{frame}\frametitle{亚像素峰值定位方法}
定义~$M$~通道：
\begin{equation}
M_i=\max(R_i,G_i,B_i)-\min(R_i,G_i,B_i)
\nonumber
\end{equation}

加权平均，但只取峰值附近的点：
\begin{equation}
\label{eq:Iestimated} I_{estimated}=
\frac{\sum{i\cdot{}M_{i}}}{\sum{i}},\textrm{ for } M_i \ge
\alpha{}M_{max}
\nonumber
\end{equation}
\end{frame}
