\subsection{技术挑战}
\begin{frame}   \frametitle{技术挑战}
\begin{itemize}
  \item<1-> 物体形状复杂，高精度高灵敏度来获取凹陷，处理遮挡
  \item<2-> 物体种类和材料多样，表面性质各异
  \item<3-> 自动选择重建方法
  \item<4-> 语义3D识别
  \item<5-> 重建尺度范围大
\end{itemize}
\end{frame}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\subsection{方法分类}
\begin{frame}   \frametitle{方法分类}
    \begin{itemize}
      \item<1-> 主动方法：光源受特别控制，帮助获取3维信息
      \item<2-> 被动方法：光源没有特别控制，只是影响图像质量
      \item<3-> 单观察点：只有一个视角
      \item<4-> 多观察点：多视角
    \end{itemize}
\end{frame}
